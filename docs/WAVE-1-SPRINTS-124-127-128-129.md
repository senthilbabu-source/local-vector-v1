## 🎯 Objective

Fix the hardcoded `dataHealth = 100` in `deriveRealityScore()` so the Reality Score reflects actual location data completeness.

**Current broken state:**
```typescript
// app/dashboard/page.tsx — line 81
```

**What we're building:** A `computeDataHealth(locationId, supabase)` service that scores 5 real dimensions of data completeness (0–100), then wires it into `deriveRealityScore()` replacing the hardcoded `100`. ## 🏗️ Architecture — What to Build

### Data Flow

```
Nightly cron (2:30 AM UTC) runs computeDataHealth(locationId, serviceSupabase)
  ↓
Writes result to locations.data_health_score (FLOAT, already in schema)
  ↓
fetchDashboardData() reads locations.data_health_score
  ↓
deriveRealityScore(openAlertCount, visibilityScore, simulationScore, dataHealthScore)
  ↓
DataHealth dimension = dataHealthScore (not hardcoded 100)
```

### Scoring Dimensions (5 total, 100 points)

| Dimension | Points | What triggers full score |
|-----------|--------|--------------------------|
| Core identity | 30 | name + address_line1 + city + phone + website_url all non-null |
| Hours completeness | 20 | hours_data JSONB has all 7 days, none null |
| Amenities coverage | 20 | ≥50% of amenities keys set to non-null (SKIP entirely if `gbp_import_source = true`) |
| Category + description | 15 | categories non-null + description/display_name ≥ 50 chars |
| Menu / services | 15 | ≥1 magic_menu row for this location with `is_published = true` |

**GBP import rule:** When `gbp_import_source = true`, the amenities dimension scores 20/20 automatically. Imported locations have intentionally null amenities (GBP doesn't export amenities). Never penalize them.

```typescript
// lib/services/data-health.service.ts — Sprint 124: DataHealth Scoring
//
// Computes a 0–100 data completeness score for a location.
// Used by deriveRealityScore() as the DataHealth factor.
//
// PURE SERVICE — never creates its own Supabase client.
// Callers pass a scoped or service-role client.
// AI_RULES §4 (services are pure), §17 (Sentry on all catch)

import * as Sentry from '@sentry/nextjs';
import type { SupabaseClient } from '@supabase/supabase-js';
import type { Database } from '@/lib/supabase/database.types';

export interface DataHealthScore {
  total: number;               // 0–100
  coreIdentity: number;        // 0–30
  hoursCompleteness: number;   // 0–20
  amenitiesCoverage: number;   // 0–20
  categoryDescription: number; // 0–15
  menuServices: number;        // 0–15
  gbpImportSource: boolean;    // whether amenities was auto-scored
}

export interface DataHealthInput {
  // From locations table:
  name: string | null;
  address_line1: string | null;
  city: string | null;
  phone: string | null;
  website_url: string | null;
  hours_data: Record<string, unknown> | null;
  amenities: Record<string, unknown> | null;
  categories: unknown[] | null;
  display_name: string | null;
  gbp_import_source: boolean;
  // From magic_menus query:
  hasPublishedMenu: boolean;
}

/**
 * Compute DataHealth score from a DataHealthInput.
 * PURE FUNCTION — no I/O, no side effects.
 * Exported for unit testing.
 */
export function scoreDataHealth(input: DataHealthInput): DataHealthScore {
  // ── Core Identity (30 pts) ────────────────────────────────────────
  const identityFields = [input.name, input.address_line1, input.city, input.phone, input.website_url];
  const identityPresent = identityFields.filter(Boolean).length;
  const coreIdentity = Math.round((identityPresent / identityFields.length) * 30);

  // ── Hours Completeness (20 pts) ──────────────────────────────────
  let hoursCompleteness = 0;
  if (input.hours_data && typeof input.hours_data === 'object') {
    const days = ['monday','tuesday','wednesday','thursday','friday','saturday','sunday'];
    const presentDays = days.filter(d => input.hours_data![d] !== null && input.hours_data![d] !== undefined).length;
    hoursCompleteness = Math.round((presentDays / 7) * 20);
  }

  // ── Amenities Coverage (20 pts) ──────────────────────────────────
  let amenitiesCoverage = 0;
  if (input.gbp_import_source) {
    // GBP-imported: never penalize for null amenities
    amenitiesCoverage = 20;
  } else if (input.amenities && typeof input.amenities === 'object') {
    const keys = Object.keys(input.amenities);
    if (keys.length > 0) {
      const nonNull = keys.filter(k => input.amenities![k] !== null && input.amenities![k] !== undefined).length;
      const coverage = nonNull / keys.length;
      amenitiesCoverage = coverage >= 0.5 ? 20 : Math.round(coverage * 40); // partial credit below 50%
    }
  }

  // ── Category + Description (15 pts) ─────────────────────────────
  let categoryDescription = 0;
  const hasCategory = Array.isArray(input.categories) && input.categories.length > 0;
  const descriptionText = input.display_name ?? '';
  const hasDescription = descriptionText.length >= 50;
  if (hasCategory) categoryDescription += 8;
  if (hasDescription) categoryDescription += 7;

  // ── Menu / Services (15 pts) ─────────────────────────────────────
  const menuServices = input.hasPublishedMenu ? 15 : 0;

  const total = coreIdentity + hoursCompleteness + amenitiesCoverage + categoryDescription + menuServices;

  return {
    total: Math.min(100, Math.max(0, total)),
    coreIdentity,
    hoursCompleteness,
    amenitiesCoverage,
    categoryDescription,
    menuServices,
    gbpImportSource: input.gbp_import_source,
  };
}

/**
 * Fetch location data and compute DataHealth score.
 * Used by the nightly cron and by fetchDashboardData().
 */
export async function computeDataHealth(
  locationId: string,
  supabase: SupabaseClient<Database>,
): Promise<DataHealthScore | null> {
  try {
    const [locResult, menuResult] = await Promise.all([
      supabase
        .from('locations')
        .select('name,address_line1,city,phone,website_url,hours_data,amenities,categories,display_name,gbp_import_source')
        .eq('id', locationId)
        .single(),
      supabase
        .from('magic_menus')
        .select('id')
        .eq('location_id', locationId)
        .eq('is_published', true)
        .limit(1),
    ]);

    if (locResult.error || !locResult.data) return null;

    const loc = locResult.data;
    const input: DataHealthInput = {
      name: loc.name,
      address_line1: loc.address_line1,
      city: loc.city,
      phone: loc.phone,
      website_url: loc.website_url,
      hours_data: loc.hours_data as Record<string, unknown> | null,
      amenities: loc.amenities as Record<string, unknown> | null,
      categories: loc.categories as unknown[] | null,
      display_name: loc.display_name,
      gbp_import_source: loc.gbp_import_source ?? false,
      hasPublishedMenu: (menuResult.data ?? []).length > 0,
    };

    return scoreDataHealth(input);
  } catch (err) {
    Sentry.captureException(err, { tags: { service: 'data-health', sprint: '124' } });
    return null;
  }
}
```

### Component 5: `app/api/cron/data-health-refresh/route.ts` — NEW

```typescript
// app/api/cron/data-health-refresh/route.ts — Sprint 124
//
// Nightly cron: recompute data_health_score for all active locations.
// Schedule: 2:30 AM UTC (after SOV cron at 2 AM, before autopilot at 2 AM Wed)
// vercel.json: { "path": "/api/cron/data-health-refresh", "schedule": "30 2 * * *" }
//
// Kill switch: DATA_HEALTH_CRON_DISABLED=true

import { NextResponse } from 'next/server';
import * as Sentry from '@sentry/nextjs';
import { createServiceRoleClient } from '@/lib/supabase/server';
import { computeDataHealth } from '@/lib/services/data-health.service';

export const runtime = 'nodejs';
export const maxDuration = 300;

export async function GET(request: Request) {
  // Kill switch
  if (process.env.DATA_HEALTH_CRON_DISABLED === 'true') {
    return NextResponse.json({ skipped: true, reason: 'kill switch active' });
  }

  // Auth
  const authHeader = request.headers.get('authorization');
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return new Response('Unauthorized', { status: 401 });
  }

  const supabase = createServiceRoleClient();

  // Fetch all active location IDs
  const { data: locations, error } = await supabase
    .from('locations')
    .select('id')
    .eq('is_archived', false);

  if (error) {
    Sentry.captureException(error, { tags: { cron: 'data-health-refresh', sprint: '124' } });
    return NextResponse.json({ error: error.message }, { status: 500 });
  }

  let updated = 0;
  let failed = 0;

  for (const loc of locations ?? []) {
    try {
      const score = await computeDataHealth(loc.id, supabase);
      if (score !== null) {
        await supabase
          .from('locations')
          .update({ data_health_score: score.total })
          .eq('id', loc.id);
        updated++;
      }
    } catch (err) {
      failed++;
      Sentry.captureException(err, { tags: { cron: 'data-health-refresh', locationId: loc.id } });
    }
  }

  return NextResponse.json({ updated, failed, total: (locations ?? []).length });
}
```

## 🧠 Edge Cases

1. **Location has no `gbp_import_source` column yet** (before migration runs): column defaults to `FALSE` — safe.
2. **`data_health_score` is NULL** in locations (before first cron run): `deriveRealityScore` falls back to `100` (neutral) — no regression.
3. **`hours_data` has unexpected shape** (e.g., nested `periods` array from GBP): check for null value per day key, not for specific shape. Treat unexpected shapes as 0 for that day.
4. **Location has 0 amenity keys** in amenities JSONB (empty `{}`): score 0 for amenities (not 20). An empty object is different from a populated one.
5. **Cron runs during high-DB load**: `for` loop with per-location `await` is intentionally sequential — no parallel explosion. Sentry on each failure, continue processing.
6. **`deriveRealityScore` called in tests with old signature**: 4th param is optional — no breaking change.
7. **`data_health_score` column doesn't exist in `database.types.ts`** yet: After migration, regenerate types with `npx supabase gen types typescript`. Add `data_health_score: number | null` and `gbp_import_source: boolean` to `locations` row type manually if regeneration not possible.

```typescript
  specialtyTags: string[],
  industry: 'medical_dental' | string,
): Array<{ '@type': 'MedicalProcedure' | 'MedicalTherapy'; name: string; description?: string }> {
  if (!specialtyTags || specialtyTags.length === 0) return [];

  const isDental = specialtyTags.some(tag =>
    Object.keys(DENTAL_PROCEDURE_CATEGORIES).some(cat =>
      tag.toLowerCase().includes(cat.toLowerCase())
    )
  );
  const catalog = isDental ? DENTAL_PROCEDURE_CATEGORIES : MEDICAL_SPECIALTY_CATEGORIES;

  const services: Array<{ '@type': 'MedicalProcedure' | 'MedicalTherapy'; name: string }> = [];

  for (const tag of specialtyTags) {
    const category = Object.keys(catalog).find(cat =>
      cat.toLowerCase() === tag.toLowerCase() || tag.toLowerCase().includes(cat.toLowerCase())
```

```typescript
//
// Validates AI-generated copy for medical/dental contexts.
// Flags claims that could constitute medical advice or create legal exposure.
// Used by content generation pipelines when isMedicalCategory()=true.
// PURE FUNCTION — no I/O.

const FORBIDDEN_PATTERNS = [
  /\bwe (diagnose|treat|cure|heal|prescribe)\b/i,
  /\bguarantee[sd]?\b/i,
  /\b100% success\b/i,
  /\bno side effects\b/i,
  /\bbest (doctor|dentist|practice) in\b/i,
  /\bpainless\b/i,         // subjective claim
];

const DISCLAIMER_REQUIRED_PATTERNS = [
  /\btreatment\b/i,
  /\bprocedure\b/i,
  /\bdiagnosis\b/i,
  /\bmedication\b/i,
];

export interface CopyGuardResult {
  approved: boolean;
  violations: string[];
  requiresDisclaimer: boolean;
  suggestionToAdd?: string;
}

export function checkMedicalCopy(text: string): CopyGuardResult {
  const violations: string[] = [];

  for (const pattern of FORBIDDEN_PATTERNS) {
    if (pattern.test(text)) {
      violations.push(`Forbidden pattern: ${pattern.toString()}`);
    }
  }

  const requiresDisclaimer = DISCLAIMER_REQUIRED_PATTERNS.some(p => p.test(text));

  return {
    approved: violations.length === 0,
    violations,
    requiresDisclaimer,
    suggestionToAdd: requiresDisclaimer
      ? 'Results may vary. Consult with our team to discuss your specific needs.'
      : undefined,
  };
}
```

```typescript
// lib/faq/faq-generator.ts — Sprint 128
//
// Generates FAQ Q&A pairs from location ground truth.
// PURE FUNCTION — no I/O. Callers fetch data and pass it in.
// AI_RULES §158 (DataHealth), §159 (medical templates)

import { createHash } from 'crypto';
import { isMedicalCategory } from '@/lib/services/sov-seed';
import {
  MEDICAL_FAQ_TEMPLATES,
  getApplicableTemplates,
  renderFAQTemplate,
} from '@/lib/services/medical-faq-templates';

export interface FAQPair {
  id: string;            // content hash (SHA-256 of question, first 12 chars)
  question: string;
  answer: string;
  contentHash: string;   // SHA-256 of question (full, for exclusion matching)
  source: 'hours' | 'location' | 'menu' | 'amenity' | 'operational' | 'medical';
}

export interface FAQGeneratorInput {
  name: string;
  city: string;
  state: string;
  phone: string | null;
  website_url: string | null;
  hours_data: Record<string, { open: string; close: string } | null> | null;
  amenities: Record<string, boolean | null> | null;
  categories: string[] | null;
  display_name: string | null;
  operational_status: string;
  // Menu data
  menuItemNames: string[];   // top 5 published menu item names
  // Medical fields (Sprint 127)
  accepting_new_patients: boolean | null;
  telehealth_available: boolean | null;
  insurance_types: string[] | null;
  specialty_tags: string[] | null;
}

function makeHash(question: string): string {
  return createHash('sha256').update(question).digest('hex');
}

function formatHours(hours_data: FAQGeneratorInput['hours_data']): string {
  if (!hours_data) return 'during regular business hours';
  const days = ['monday','tuesday','wednesday','thursday','friday','saturday','sunday'];
  const open = days.filter(d => hours_data[d] !== null && hours_data[d] !== undefined);
  if (open.length === 7) return 'Monday–Sunday';
  if (open.length === 5 && !open.includes('saturday') && !open.includes('sunday')) return 'Monday–Friday';
  return open.map(d => d.charAt(0).toUpperCase() + d.slice(1, 3)).join(', ');
}

/**
 * Generate FAQ pairs from location ground truth.
 * Returns 3–15 pairs. Never generates for null fields.
 * PURE FUNCTION — no side effects.
 */
export function generateFAQs(input: FAQGeneratorInput): FAQPair[] {
  const pairs: FAQPair[] = [];
  const isMedical = isMedicalCategory(input.categories ?? []);

  // ── Hours ─────────────────────────────────────────────────────────
  if (input.hours_data) {
    const hoursStr = formatHours(input.hours_data);
    const q = `What are ${input.name}'s hours?`;
    const a = `${input.name} in ${input.city}, ${input.state} is open ${hoursStr}.${input.phone ? ` Call ${input.phone} for holiday hours.` : ''}`;
    const hash = makeHash(q);
    pairs.push({ id: hash.slice(0, 12), question: q, answer: a, contentHash: hash, source: 'hours' });
  }

  // ── Location ──────────────────────────────────────────────────────
  if (input.city && input.state) {
    const q = `Where is ${input.name} located?`;
    const a = `${input.name} is located in ${input.city}, ${input.state}.${input.phone ? ` Call ${input.phone} for directions.` : ''}`;
    const hash = makeHash(q);
    pairs.push({ id: hash.slice(0, 12), question: q, answer: a, contentHash: hash, source: 'location' });
  }

  // ── Phone / Contact ───────────────────────────────────────────────
  if (input.phone) {
    const q = `How do I contact ${input.name}?`;
    const a = `Call ${input.name} at ${input.phone}.${input.website_url ? ` You can also visit ${input.website_url} for more information.` : ''}`;
    const hash = makeHash(q);
    pairs.push({ id: hash.slice(0, 12), question: q, answer: a, contentHash: hash, source: 'location' });
  }

  // ── Operational Status ────────────────────────────────────────────
  if (input.operational_status === 'OPERATIONAL') {
    const q = `Is ${input.name} still open?`;
    const a = `Yes, ${input.name} in ${input.city} is currently open and operating.`;
    const hash = makeHash(q);
    pairs.push({ id: hash.slice(0, 12), question: q, answer: a, contentHash: hash, source: 'operational' });
  }

  // ── Menu Items ────────────────────────────────────────────────────
  if (!isMedical && input.menuItemNames.length > 0) {
    const topItems = input.menuItemNames.slice(0, 3).join(', ');
    const q = `What does ${input.name} serve?`;
    const a = `${input.name} serves items including ${topItems}. Visit our menu page for the full selection.`;
    const hash = makeHash(q);
    pairs.push({ id: hash.slice(0, 12), question: q, answer: a, contentHash: hash, source: 'menu' });
  }

  // ── Amenities ─────────────────────────────────────────────────────
  if (!isMedical && input.amenities) {
    const trueAmenities = Object.entries(input.amenities)
      .filter(([, v]) => v === true)
      .map(([k]) => k.replace(/_/g, ' '))
      .slice(0, 3);

    if (trueAmenities.length > 0) {
      const q = `Does ${input.name} have ${trueAmenities[0]}?`;
      const a = `Yes, ${input.name} in ${input.city} offers ${trueAmenities.join(', ')}.`;
      const hash = makeHash(q);
      pairs.push({ id: hash.slice(0, 12), question: q, answer: a, contentHash: hash, source: 'amenity' });
    }
  }

  // ── Medical Templates (Sprint 127) ────────────────────────────────
  if (isMedical) {
    const locationData: Record<string, unknown> = {
      accepting_new_patients: input.accepting_new_patients,
      telehealth_available: input.telehealth_available,
      insurance_types: input.insurance_types,
      specialty_tags: input.specialty_tags,
      phone: input.phone,
      hours_data: input.hours_data,
    };
    const applicable = getApplicableTemplates(MEDICAL_FAQ_TEMPLATES, locationData);
    const insuranceList = input.insurance_types?.join(', ') ?? undefined;

    for (const template of applicable.slice(0, 8)) {
      const rendered = renderFAQTemplate(template, {
        businessName: input.name,
        city: input.city,
        phone: input.phone,
        insuranceList,
        hoursString: formatHours(input.hours_data),
      });
      if (rendered) {
        const hash = makeHash(rendered.question);
        pairs.push({
          id: hash.slice(0, 12),
          question: rendered.question,
          answer: rendered.answer,
          contentHash: hash,
          source: 'medical',
        });
      }
    }
  }

  // Cap at 15 pairs
  return pairs.slice(0, 15);
}

/**
 * Filter out excluded pairs by content hash.
 */
export function applyExclusions(pairs: FAQPair[], excludedHashes: string[]): FAQPair[] {
  const excluded = new Set(excludedHashes);
  return pairs.filter(p => !excluded.has(p.contentHash));
}
```

```typescript
// lib/faq/faq-schema-builder.ts — Sprint 128
// Serializes FAQPair[] to FAQPage JSON-LD string.
// Caps at 10 pairs. Strips HTML. Validates answer length.
// PURE FUNCTION.

import type { FAQPair } from './faq-generator';

const MAX_INJECTION_PAIRS = 10;
const MAX_ANSWER_CHARS = 300;

function stripHtml(text: string): string {
  return text.replace(/<[^>]+>/g, '').replace(/&amp;/g, '&').replace(/&lt;/g, '<').replace(/&gt;/g, '>').trim();
}

function truncateAnswer(answer: string): string {
  const clean = stripHtml(answer);
  return clean.length > MAX_ANSWER_CHARS
    ? clean.slice(0, MAX_ANSWER_CHARS - 3) + '...'
    : clean;
}

export function toFAQPageJsonLd(pairs: FAQPair[]): string {
  const capped = pairs.slice(0, MAX_INJECTION_PAIRS);

  const schema = {
    '@context': 'https://schema.org',
    '@type': 'FAQPage',
    mainEntity: capped.map(pair => ({
      '@type': 'Question',
      name: stripHtml(pair.question),
      acceptedAnswer: {
        '@type': 'Answer',
        text: truncateAnswer(pair.answer),
      },
    })),
  };

  return JSON.stringify(schema);
}
```
